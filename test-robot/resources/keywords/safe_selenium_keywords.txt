*** Keywords ***
Safe Input Text
    [Arguments]    ${field}    ${text}    ${error}=${None}
    Wait Until Page Contains Element    ${field}    timeout=5    error=${error}
    Input Text    ${field}    ${text}

Safe Click Element
    [Arguments]    ${element}    ${error}=${None}
    Wait Until Page Contains Element    ${element}    timeout=5    error=${error}
    Click Element    ${element}

Safe Click Link
    [Arguments]    ${link}    ${error}=${None}
    Wait Until Page Contains Element    ${link}    timeout=5    error=${error}
    Click Link    ${link}

Safe Select From List
    [Arguments]    ${listbox}    ${selection}    ${error}=${None}
    Wait Until Page Contains Element    ${listbox}    timeout=5    error=${error}
    Select From List    ${listbox}    ${selection}
